import type { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: "*",
        disallow: [
          "/login",
          "/register",
          "/dashboard",
          "/dashboard/*",
        ],
      },
    ],
    sitemap: "https://himna-marketing.vercel.app/sitemap.xml",
  };
}
